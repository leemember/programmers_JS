### CODE

```
function solution(cards1, cards2, goal) {
    var answer = 'Yes';

    // 단어 배열 goal만큼 for 반복문을 사용하여 shift로 단어를 차례대로 추출하여
    // 결론적으로 원하는 순서의 단어 배열을 만들 수 있다.
    for(const card of goal) {
        if(cards1[0] === card) {
            // 제거된 단어 추출
            cards1.shift();
        } else if(cards2[0] === card) {
            cards2.shift();
        } else {
            return "No"
        }
    }


    return answer;
}
```
